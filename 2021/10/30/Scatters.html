from dash.dr.scatter import mixture_scatter, within_scatter, between_scatter
import numpy as np
from pathlib import Path
import matplotlib.pyplot as plt

Step 1: Load Data

Loading the Iris Data.

data_dir = Path("datasets/iris")
X = np.loadtxt(data_dir / "iris.data", delimiter=",")
y = np.loadtxt(data_dir / "iris.labels", delimiter=",").reshape(-1, 1)

print(f"X shape: {X.shape}")
print(f"y shape: {y.shape}")

X shape: (150, 4)
y shape: (150, 1)

Step 2: The Mixture Class Scatter

Can we minimize and maximize the mixture scatter?

from dash.dr.eig import eigvector
from dash.viz.scatter import scatter2d

M = mixture_scatter(X)
v = eigvector(M, output="max", k=2)
result = (v.T @ X.T).T

# Plot
scatter2d(result, y)
